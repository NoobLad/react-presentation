Composants ?

Morceaux d'interface utilisateur réutilisable dont on se fiche de la mécanique interne.

Les promesses de React.js

  • Il gère la mise à jour du DOM
  • Ne fait rien d'autre que le rendu
  • Performance sans s'en soucier
  • Réutilisabilité
  • Tourne partout
  • Facile à tester

Un composant ça ressemble à ça

SimpleComponent.jsx 

    function SimpleComponent() {
        return (
            <div>
                <h1>Hello</h1>
                <hr/>
            </div>
        );
    }

Le même en JS


    function SimpleComponent() {
        return React.createElement(
            "div",
            null,
            React.createElement(
                "h1",
                null,
                "Hello"
            ),
            React.createElement("hr", null)
        );
    }

Notre composant

Afficher une variable


    function HelloWorld() {
        var world = "WOORRRLLLLDDDDD";
        return (
            <h1>Hello {world} !</h1>
        );
    }

Afficher dans le document


    React.render(<HelloWorld />, document.querySelector('.hello-container'));

Avec une classe


    class HelloWorld extends React.Component {
        render() {
            var world = "WOORRRLLLLDDDDD";
            return (
                <h1>Hello {world} !</h1>
            );
        }
    }

State


    class Counter extends React.Component {
        constructor(props, context) {
            super(props, context);
            this.state = {
                counter: 0
            };
        }
        render() {
            return (
                <div>
                    <p>{this.state.counter}</p>
                </div>
            );
        }
    }

Listeners


    class Counter extends React.Component {
        constructor(props, context) {
            super(props, context);
            this.state = {
                counter: 0
            };
        }
        increase() {
            this.setState({
                counter: this.state.counter + 1
            });
        }
        render() {
            return (
                <div>
                    <p>{this.state.counter}</p>
                    <button type="button" onClick={() => this.increase()}>+</button>
                </div>
            );
        }
    }

Composer des composants


    function CounterButton({onIncrease}) {
        return (
            <button type="button" onClick={onIncrease}>+</button>
        );
    }

    function CounterDisplay({counter}) {
        return (
            <p>{counter}</p>
        );
    }

    class Counter extends React.Component {
        constructor(props, context) {
            super(props, context);
            this.state = {
                counter: 0
            };
        }
        increase() {
            this.setState({
                counter: this.state.counter + 1
            });
        }
        render() {
            return (
                <div>
                    <CounterDisplay counter={this.state.counter} />
                    <CounterButton onIncrease={() => this.increase()} />
                </div>
            );
        }
    }

Conditions


    // Imaginez que ça puisse changer de valeur :p
    let show = true;
    var world = "WOORRRLLLLDDDDD";

    function HelloWorld1() {
        return (
            <h1>Hello { show && world} !</h1>
        );
    }

    function HelloWorld2() {
        return (
            <h1>Hello { show ? world : ""} !</h1>
        );
    }

    function HelloWorld3() {
        const showWorld = () => {
            if (show) return world;
            return "";
        }
        return (
            <h1>Hello {showWorld()} !</h1>
        );
    }

                {text}

        );
    }
